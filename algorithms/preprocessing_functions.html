Question¶

Complete the following function to split data between training and test sets, according to the test_size ratio.

def train_test_split(data, test_size):
    limit = int(len(data)*(1-test_size))
    return data[:limit], data[limit:]

# Generate a random (30,3) tensor
x = np.random.rand(30, 3)

x_train, x_test = train_test_split(x, .25)
print(f'x_train: {x_train.shape}. x_test: {x_test.shape}')

def scale_min_max(data):

Question¶

Complete the following function to implement feature scaling by standardization.

def scale_standard(data, mean=None, std=None):
    if mean and std:
        return (data - mean) / std
    else:
        return (data - np.mean(data)) / np.std(data)

# Generate a random (3,3) tensor with values between 1 and 10
x = np.random.randint(1, 10, (3,3))

x_scaled = scale_standard(x)
print(x_scaled)

mean_scaled = x_scaled.mean()
std_scaled = x_scaled.std()
print(f'Mean: {mean_scaled}. Std: {std_scaled}')

assert np.abs(np.max(mean_scaled)) < 10**-6
assert np.abs(np.max(std_scaled - 1)) < 10**-6

def one_hot_encode(elements):
    result = np.zeros((len(elements), max(elements) + 1))
    for i, element in enumerate(elements):
        result[i, element] = 1
    
    return result
